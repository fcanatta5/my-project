# Instalação e configuração do Btrfs + Snapper no Arch Linux
#
# OBSERVAÇÃO IMPORTANTE:
# - Este roteiro assume que o sistema já está instalado em Btrfs e que a raiz / usa um subvolume (ex.: @).
# - Se o seu Arch estiver em ext4/xfs, a migração para Btrfs é outro processo e não está coberta aqui.
# - Rode os comandos como root ou com sudo.

# 1) Confirmar que a raiz está em Btrfs
findmnt -no FSTYPE /
findmnt -no SOURCE /
sudo btrfs subvolume list /

# 2) Instalar os pacotes principais
sudo pacman -Syu --needed btrfs-progs snapper

# 3) Opcional, mas recomendado:
# - snap-pac: cria snapshots pre/post em transações do pacman
# - grub-btrfs + inotify-tools: adiciona snapshots ao menu do GRUB
sudo pacman -S --needed snap-pac grub-btrfs inotify-tools

# 4) Se /.snapshots já existir montado (comum em layouts do archinstall), desmontar e remover o ponto de montagem
sudo umount /.snapshots 2>/dev/null || true
sudo rm -rf /.snapshots

# 5) Criar a configuração principal do Snapper para a raiz /
#    Isso cria temporariamente o subvolume /.snapshots
sudo snapper -c root create-config /

# 6) Remover o subvolume /.snapshots criado automaticamente pelo Snapper
#    para recriá-lo como um subvolume dedicado de nível superior (@snapshots)
sudo btrfs subvolume delete /.snapshots
sudo mkdir -p /.snapshots

# 7) Descobrir o device e UUID do sistema de arquivos raiz
ROOT_DEV="$(findmnt -no SOURCE /)"
ROOT_UUID="$(findmnt -no UUID /)"
echo "$ROOT_DEV"
echo "$ROOT_UUID"

# 8) Montar o top-level do Btrfs (subvolid=5) e criar o subvolume dedicado @snapshots
sudo mkdir -p /mnt/btrfs-top
sudo mount -o subvolid=5 "$ROOT_DEV" /mnt/btrfs-top
sudo btrfs subvolume create /mnt/btrfs-top/@snapshots
sudo umount /mnt/btrfs-top
sudo rmdir /mnt/btrfs-top

# 9) Adicionar o subvolume @snapshots ao /etc/fstab
#    Ajuste as opções de montagem se você já usa outras opções no subvolume raiz.
echo "UUID=$ROOT_UUID /.snapshots btrfs subvol=@snapshots,compress=zstd,noatime 0 0" | sudo tee -a /etc/fstab

# 10) Montar tudo novamente
sudo mount -a
findmnt /.snapshots

# 11) Ajustar a política do Snapper para snapshots automáticos + limpeza
sudo snapper -c root set-config "TIMELINE_CREATE=yes TIMELINE_CLEANUP=yes NUMBER_CLEANUP=yes EMPTY_PRE_POST_CLEANUP=yes NUMBER_LIMIT=10 NUMBER_LIMIT_IMPORTANT=10 TIMELINE_LIMIT_HOURLY=10 TIMELINE_LIMIT_DAILY=7 TIMELINE_LIMIT_WEEKLY=4 TIMELINE_LIMIT_MONTHLY=3 TIMELINE_LIMIT_YEARLY=0"

# 12) (Opcional) Permitir que seu usuário administre snapshots sem usar root
#     Troque SEU_USUARIO pelo nome do seu usuário.
sudo snapper -c root set-config "ALLOW_USERS=SEU_USUARIO SYNC_ACL=yes"

# 13) Ativar timers automáticos do Snapper
sudo systemctl enable --now snapper-timeline.timer
sudo systemctl enable --now snapper-cleanup.timer

# 14) (Opcional) Ativar integração com o GRUB para snapshots bootáveis no menu
sudo systemctl enable --now grub-btrfsd.service

# 15) Verificar a configuração
sudo snapper list-configs
sudo snapper -c root get-config
sudo snapper -c root list
systemctl status snapper-timeline.timer snapper-cleanup.timer --no-pager

# 16) Criar um snapshot manual de teste
sudo snapper -c root create -d "snapshot manual inicial"
sudo snapper -c root list

# 17) Comandos úteis do dia a dia
# Listar snapshots
sudo snapper -c root list

# Comparar mudanças entre snapshots (exemplo: snapshot 10 e 11)
sudo snapper -c root status 10..11
sudo snapper -c root diff 10..11

# Criar snapshot manual
sudo snapper -c root create -d "antes de atualizar"

# Limpeza manual
sudo snapper -c root cleanup number
sudo snapper -c root cleanup timeline

# 18) (Opcional) Ativar quota/qgroups para limpeza baseada em espaço
#     Só faça isso se você realmente quiser controle de limpeza por uso de espaço.
sudo snapper -c root setup-quota

# 19) (Opcional) Configurar limites de espaço após ativar quota
sudo snapper -c root set-config "SPACE_LIMIT=0.5 FREE_LIMIT=0.2"

# 20) Reiniciar no final
sudo reboot
